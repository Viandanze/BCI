```javascript
// src/scene/ThreeScene.js
import * as THREE from 'three';
import { OrbitControls } from 'three/addons/controls/OrbitControls.js';
import { EffectComposer } from 'three/addons/postprocessing/EffectComposer.js';
import { RenderPass } from 'three/addons/postprocessing/RenderPass.js';
import { UnrealBloomPass } from 'three/addons/postprocessing/UnrealBloomPass.js';
import { OutputPass } from 'three/addons/postprocessing/OutputPass.js';

export class DigitalTwinScene {
    constructor(container) {
        this.container = container;
        this.models = new Map(); // 设备ID → Three.js对象
        this.init();
    }

    init() {
        // 1. 场景
        this.scene = new THREE.Scene();
        this.scene.background = new THREE.Color(0x0a0a0f);

        // 2. 相机
        this.camera = new THREE.PerspectiveCamera(
            60, 
            this.container.clientWidth / this.container.clientHeight,
            0.1, 
            1000
        );
        this.camera.position.set(10, 8, 10);

        // 3. 渲染器（优先WebGPU，降级WebGL）
        this.renderer = this.initRenderer();

        // 4. 控制器
        this.controls = new OrbitControls(this.camera, this.renderer.domElement);
        this.controls.enableDamping = true;

        // 5. 灯光（PBR需要HDRI环境光）
        this.setupLights();

        // 6. 后期处理（工业感渲染）
        this.setupPostProcessing();

        // 7. 网格地面（参考网格）
        this.setupGround();

        // 8. 响应式
        window.addEventListener('resize', () => this.onResize());
    }

    initRenderer() {
        // 2026主流：尝试WebGPU，降级WebGL
        try {
            const renderer = new THREE.WebGPURenderer({
                antialias: true,
                powerPreference: 'high-performance'
            });
            renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
            console.log('✅ WebGPU渲染器启动');
            return renderer;
        } catch (e) {
            console.log('⚠️ WebGPU不可用，回退WebGL2');
            const renderer = new THREE.WebGLRenderer({
                antialias: true,
                powerPreference: 'high-performance'
            });
            renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
            return renderer;
        }
    }

    setupLights() {
        // 环境光
        const ambient = new THREE.AmbientLight(0x404040, 0.5);
        this.scene.add(ambient);

        // 主方向光（模拟太阳）
        const sun = new THREE.DirectionalLight(0xffffff, 1.5);
        sun.position.set(10, 20, 10);
        sun.castShadow = true; // 工业场景开阴影
        this.scene.add(sun);

        // 补光
        const fill = new THREE.DirectionalLight(0x8080ff, 0.3);
        fill.position.set(-10, 5, -10);
        this.scene.add(fill);
    }

    setupPostProcessing() {
        this.composer = new EffectComposer(this.renderer);
        
        // 基础渲染
        const renderPass = new RenderPass(this.scene, this.camera);
        this.composer.addPass(renderPass);

        // 辉光效果（工业设备常用）
        const bloomPass = new UnrealBloomPass(
            new THREE.Vector2(window.innerWidth, window.innerHeight),
            0.8,  // 强度
            0.4,  // 半径
            0.85  // 阈值
        );
        this.composer.addPass(bloomPass);

        // 输出
        const outputPass = new OutputPass();
        this.composer.addPass(outputPass);
    }

    setupGround() {
        const grid = new THREE.GridHelper(20, 20, 0x303030, 0x202020);
        this.scene.add(grid);
    }

    animate() {
        requestAnimationFrame(() => this.animate());
        this.controls.update();
        this.composer.render();
    }

    onResize() {
        const w = this.container.clientWidth;
        const h = this.container.clientHeight;
        this.camera.aspect = w / h;
        this.camera.updateProjectionMatrix();
        this.renderer.setSize(w, h);
        this.composer.setSize(w, h);
    }
}
```

### 阶段三：GLTF模型加载 + 数据绑定（40分钟）

```javascript
// src/components/DeviceModel.js
import * as THREE from 'three';
import { GLTFLoader } from 'three/addons/loaders/GLTFLoader.js';
import { DRACOLoader } from 'three/addons/loaders/DRACOLoader.js';
import gsap from 'gsap';

export class DeviceModel {
    constructor(scene, deviceId, position) {
        this.scene = scene;
        this.deviceId = deviceId;
        this.position = position;
        this.mesh = null;
        this.originalColor = null;
        
        // 数据映射：传感器字段 → 3D属性
        this.dataMapping = {
            'temperature': { prop: 'emissive', colorRange: [0x0000ff, 0xff0000] },
            'humidity': { prop: 'scale', range: [0.8, 1.2] },
            'distance': { prop: 'rotation.y', animated: true }
        };
    }

    async loadModel(url) {
        const dracoLoader = new DRACOLoader();
        dracoLoader.setDecoderPath('https://www.gstatic.com/draco/v1/decoders/');

        const loader = new GLTFLoader();
        loader.setDRACOLoader(dracoLoader);

        return new Promise((resolve, reject) => {
            loader.load(
                url,
                (gltf) => {
                    this.mesh = gltf.scene;
                    this.mesh.position.copy(this.position);
                    this.mesh.userData.deviceId = this.deviceId;

                    // 遍历所有材质，设置PBR
                    this.mesh.traverse((child) => {
                        if (child.isMesh) {
                            child.castShadow = true;
                            child.receiveShadow = true;
                            
                            if (child.material) {
                                // 保存原始颜色
                                if (child.material.color) {
                                    this.originalColor = child.material.color.clone();
                                }
                            }
                        }
                    });

                    this.scene.add(this.mesh);
                    resolve(this.mesh);
                },
                undefined,
                reject
            );
        });
    }

    updateFromSensor(data) {
        if (!this.mesh) return;

        // 温度 → 发光颜色（蓝→红渐变）
        if (data.temperature !== undefined) {
            const temp = data.temperature;
            const normalized = Math.max(0, Math.min(1, (temp - 15) / 35)); // 15°C蓝，50°C红
            const color = new THREE.Color();
            
            if (normalized < 0.5) {
                // 蓝色→黄色
                color.setHex(0x0066ff).lerp(new THREE.Color(0xffff00), normalized * 2);
            } else {
                // 黄色→红色
                color.setHex(0xffff00).lerp(new THREE.Color(0xff0000), (normalized - 0.5) * 2);
            }

            this.mesh.traverse((child) => {
                if (child.isMesh && child.material) {
                    // 发光效果
                    child.material.emissive = color;
                    child.material.emissiveIntensity = 0.3 + normalized * 0.5;
                }
            });
        }

        // 湿度 → 模型缩放
        if (data.humidity !== undefined) {
            const scale = 0.8 + (data.humidity / 100) * 0.4;
            gsap.to(this.mesh.scale, {
                x: scale, y: scale, z: scale,
                duration: 0.5,
                ease: 'power2.out'
            });
        }

        // 距离 → 旋转动画
        if (data.distance !== undefined && data.distance < 30) {
            gsap.to(this.mesh.rotation, {
                y: this.mesh.rotation.y + Math.PI * 0.1,
                duration: 0.3
            });
        }

        // 告警特效
        if (data.alert) {
            this.triggerAlert(data.alert);
        }
    }

    triggerAlert(alertType) {
        const colors = {
            'warning': 0xffff00,
            'danger': 0xff0000,
            'info': 0x00ffff
        };

        // 脉冲动画
        gsap.to(this.mesh.scale, {
            x: 1.1, y: 1.1, z: 1.1,
            duration: 0.2,
            yoyo: true,
            repeat: 3,
            onComplete: () => {
                this.mesh.scale.set(1, 1, 1);
            }
        });

        // 发光闪烁
        this.mesh.traverse((child) => {
            if (child.isMesh && child.material) {
                gsap.to(child.material, {
                    emissiveIntensity: 1.5,
                    duration: 0.1,
                    yoyo: true,
                    repeat: 3
                });
            }
        });
    }
}
```

```javascript
// src/utils/DataBridge.js
import mqtt from 'mqtt';

export class SensorDataBridge {
    constructor(scene) {
        this.scene = scene;
        this.devices = new Map();
        this.mqttClient = null;
        this.reconnectAttempts = 0;
        this.maxReconnectAttempts = 5;
    }

    connect(brokerUrl = 'ws://localhost:9001') {
        this.mqttClient = mqtt.connect(brokerUrl, {
            clientId: `digital_twin_${Date.now()}`,
            clean: true,
            reconnectPeriod: 5000
        });

        this.mqttClient.on('connect', () => {
            console.log('✅ MQTT连接成功');
            this.mqttClient.subscribe('sensor/#', { qos: 1 });
        });

        this.mqttClient.on('message', (topic, message) => {
            this.handleMessage(topic, message);
        });

        this.mqttClient.on('error', (err) => {
            console.error('❌ MQTT错误:', err.message);
        });

        this.mqttClient.on('reconnect', () => {
            this.reconnectAttempts++;
            console.log(`🔄 重连中... (${this.reconnectAttempts}/${this.maxReconnectAttempts})`);
        });
    }

    handleMessage(topic, message) {
        try {
            // topic格式: sensor/arduino_01
            const deviceId = topic.split('/')[1];
            const data = JSON.parse(message.toString());

            // 发送到对应设备模型
            const device = this.devices.get(deviceId);
            if (device) {
                device.updateFromSensor(data);

                // 记录到状态
                device.lastData = data;
                device.lastUpdate = Date.now();
            }
        } catch (e) {
            console.error('解析MQTT消息失败:', e);
        }
    }

    registerDevice(deviceId, deviceModel) {
        this.devices.set(deviceId, deviceModel);
    }

    getDeviceStatus(deviceId) {
        const device = this.devices.get(deviceId);
        if (!device) return null;
        
        return {
            deviceId,
            lastData: device.lastData,
            lastUpdate: device.lastUpdate,
            online: Date.now() - device.lastUpdate < 10000
        };
    }

    disconnect() {
        if (this.mqttClient) {
            this.mqttClient.end();
        }
    }
}
```

```javascript
// src/utils/PerformanceOptimizer.js
export class PerformanceOptimizer {
    constructor(renderer, scene, camera) {
        this.renderer = renderer;
        this.scene = scene;
        this.camera = camera;
        this.stats = {
            fps: 60,
            drawCalls: 0,
            triangles: 0
        };
    }

    // LOD系统：根据距离切换模型精度
    setupLOD() {
        const lod = new THREE.LOD();
        
        // 添加不同精度的模型
        // lod.addLevel(highDetailMesh, 0);      // 近距离：高清
        // lod.addLevel(mediumDetailMesh, 50);   // 中距离：中等
        // lod.addLevel(lowDetailMesh, 100);    // 远距离：低精度
        
        return lod;
    }

    // 实例化渲染：大量相同物体
    createInstancedMeshes(count = 100) {
        const geometry = new THREE.BoxGeometry(0.1, 0.1, 0.1);
        const material = new THREE.MeshStandardMaterial({ color: 0x00ff88 });
        
        const mesh = new THREE.InstancedMesh(geometry, material, count);
        
        // 设置变换矩阵
        const matrix = new THREE.Matrix4();
        const position = new THREE.Vector3();
        
        for (let i = 0; i < count; i++) {
            position.set(
                (Math.random() - 0.5) * 10,
                Math.random() * 5,
                (Math.random() - 0.5) * 10
            );
            matrix.setPosition(position);
            mesh.setMatrixAt(i, matrix);
        }
        
        mesh.instanceMatrix.needsUpdate = true;
        return mesh;
    }

    // 纹理压缩加载
    async loadCompressedTexture(url) {
        const loader = new THREE.KTX2Loader();
        loader.setTranscoderPath('lib/basis/');
        
        const texture = await loader.loadAsync(url);
        texture.colorSpace = THREE.SRGBColorSpace;
        return texture;
    }

    // 视锥体剔除（Three.js默认开启）
    enableFrustumCulling() {
        this.scene.traverse((obj) => {
            if (obj.frustumCulled !== undefined) {
                obj.frustumCulled = true;
            }
        });
    }

    // 监控帧率
    startMonitoring() {
        let lastTime = performance.now();
        let frames = 0;

        const check = () => {
            frames++;
            const now = performance.now();
            
            if (now - lastTime >= 1000) {
                this.stats.fps = frames;
                this.stats.drawCalls = this.renderer.info.render.calls;
                this.stats.triangles = this.renderer.info.render.triangles;
                
                console.log(`FPS: ${this.stats.fps} | DrawCalls: ${this.stats.drawCalls} | Triangles: ${this.stats.triangles}`);
                
                frames = 0;
                lastTime = now;
            }

            requestAnimationFrame(check);
        };
        
        requestAnimationFrame(check);
    }
}
```

```javascript
// src/main.js
import { DigitalTwinScene } from './scene/ThreeScene.js';
import { DeviceModel } from './components/DeviceModel.js';
import { SensorDataBridge } from './utils/DataBridge.js';
import { PerformanceOptimizer } from './utils/PerformanceOptimizer.js';

async function init() {
    const container = document.getElementById('canvas-container');
    
    // 1. 初始化3D场景
    const scene = new DigitalTwinScene(container);
    
    // 2. 加载设备模型
    const device1 = new DeviceModel(scene.scene, 'arduino_01', new THREE.Vector3(0, 0, 0));
    await device1.loadModel('/models/sensor_unit.glb');
    
    const device2 = new DeviceModel(scene.scene, 'arduino_02', new THREE.Vector3(3, 0, 0));
    await device2.loadModel('/models/sensor_unit.glb');
    
    // 3. 连接数据桥
    const bridge = new SensorDataBridge(scene);
    bridge.registerDevice('arduino_01', device1);
    bridge.registerDevice('arduino_02', device2);
    bridge.connect('ws://localhost:9001');
    
    // 4. 性能监控
    const optimizer = new PerformanceOptimizer(scene.renderer, scene.scene, scene.camera);
    optimizer.startMonitoring();
    
    // 5. 开始渲染循环
    scene.animate();
    
    // 6. UI面板初始化（HTML/CSS）
    initDashboard(bridge);
}

function initDashboard(bridge) {
    // 创建状态面板
    const panel = document.createElement('div');
    panel.className = 'dashboard-panel';
    panel.innerHTML = `
        <h3>设备状态</h3>
        <div id="device-list"></div>
        <div id="fps-counter">FPS: --</div>
    `;
    document.body.appendChild(panel);
    
    // 定时更新UI
    setInterval(() => {
        const list = document.getElementById('device-list');
        let html = '';
        
        bridge.devices.forEach((device, id) => {
            const status = bridge.getDeviceStatus(id);
            const online = status?.online ? '🟢' : '🔴';
            const temp = status?.lastData?.temperature?.toFixed(1) || '--';
            
            html += `<div>${online} ${id}: ${temp}°C</div>`;
        });
        
        list.innerHTML = html;
    }, 1000);
}

init();
```

**Q: 如果传感器数据更新频率是100Hz，你怎么在Three.js中高效渲染？**

A: 
1. **批量更新**：使用BufferGeometry直接操作顶点数据，避免每次创建新对象
2. **节流渲染**：100Hz数据用30Hz渲染，中间值用插值过渡
3. **Web Worker**：数据解析放Worker线程，主线程只负责渲染
4. **实例化**：大量相同传感器用InstancedMesh减少DrawCall
5. **补充**：如果GPU性能足够，可以直接订阅全部数据用GSAP做平滑过渡
