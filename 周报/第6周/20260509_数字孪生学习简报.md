### InfluxDB：时序数据为什么不用MySQL？

| 维度 | MySQL | InfluxDB | 你的场景 |
|------|-------|----------|---------|
| **写入** | 逐行INSERT | 批量追加，10万点/秒 | 传感器2Hz持续写入 |
| **存储** | 行存储 | 列存储+LZ4压缩，10:1压缩比 | 7天数据从700MB→70MB |
| **查询** | SELECT + GROUP BY | 原生时间窗口聚合 | "过去1小时平均温度"毫秒级返回 |
| **过期** | 手动DELETE | Retention Policy自动过期 | 7天后自动删除原始数据 |
| **降采样** | 需写定时任务 | Continuous Query自动聚合 | 1分钟粒度→1小时粒度自动生成 |

**核心数据模型**：

```
InfluxDB数据结构（Line Protocol格式）

measurement,tag_key=tag_val field_key=field_val timestamp

示例：
sensor_data,device=arduino_01,location=desk temperature=25.3,humidity=60.2,pressure=1013.2,distance=15.6 1715227200000000000
           ↑ tags(索引，快速查询)     ↑ fields(实际值)                ↑ 纳秒时间戳
```

### 阶段二：InfluxDB写入客户端（30分钟）

```python
#!/usr/bin/env python3
"""
influx_writer.py - InfluxDB时序数据写入服务
功能：订阅MQTT滤波数据 → 写入InfluxDB → 自动降采样
"""

import json
import time
from datetime import datetime
from influxdb_client import InfluxDBClient, Point, WritePrecision
from influxdb_client.client.write_api import SYNCHRONOUS

# ====== InfluxDB配置 ======
INFLUX_URL = "http://localhost:8086"
INFLUX_TOKEN = "你的Token"  # 从Web界面获取
INFLUX_ORG = "digital_twin"
INFLUX_BUCKET = "sensor_data"

# ====== 初始化InfluxDB客户端 ======
client = InfluxDBClient(url=INFLUX_URL, token=INFLUX_TOKEN, org=INFLUX_ORG)
write_api = client.write_api(write_options=SYNCHRONOUS)
query_api = client.query_api()


def write_sensor_data(data: dict):
    """
    将传感器数据写入InfluxDB
    data格式: {"t": 25.3, "h": 60.2, "p": 1013.2, "d": 15.6, "ts": 12345}
    """
    point = Point("sensor_data") \
        .tag("device", "arduino_01") \
        .tag("location", "desk") \
        .field("temperature", float(data.get("t", 0))) \
        .field("humidity", float(data.get("h", 0))) \
        .field("pressure", float(data.get("p", 0))) \
        .field("distance", float(data.get("d", 0))) \
        .time(datetime.utcnow(), WritePrecision.MS)
    
    write_api.write(bucket=INFLUX_BUCKET, record=point)
    print(f"[InfluxDB] 写入: T={data.get('t')}°C H={data.get('h')}% P={data.get('p')}hPa D={data.get('d')}cm")


def query_history(measurement="sensor_data", field="temperature", 
                  window="1h", duration="-24h"):
    """
    查询历史聚合数据
    示例：过去24小时，每小时平均温度
    """
    flux_query = f'''
    from(bucket: "{INFLUX_BUCKET}")
      |> range(start: {duration})
      |> filter(fn: (r) => r._measurement == "{measurement}")
      |> filter(fn: (r) => r._field == "{field}")
      |> aggregateWindow(every: {window}, fn: mean, createEmpty: false)
      |> yield(name: "mean")
    '''
    
    tables = query_api.query(flux_query, org=INFLUX_ORG)
    results = []
    for table in tables:
        for record in table.records:
            results.append({
                "time": record.get_time().isoformat(),
                "value": round(record.get_value(), 2)
            })
    return results


# ====== Retention Policy & Continuous Query ======
def setup_retention_policy():
    """
    设置数据保留策略
    - raw_bucket: 原始数据保留7天
    - downsampled: 降采样数据保留1年
    """
    # 创建7天自动过期的bucket（原始数据）
    delete_api = client.delete_api()
    
    # InfluxDB v2使用bucket级别retention
    # 在Web界面: Data > Buckets > sensor_data > Settings > Delete Data
    # 设置Retention Period为 7d
    
    # 降采样bucket（1年保留）
    # 在Web界面: Data > Buckets > 创建 "sensor_downsampled" > Retention: 365d
    print("[配置] 请在InfluxDB Web界面设置:")
    print("  1. sensor_data bucket → Retention: 7d")
    print("  2. 创建 sensor_downsampled bucket → Retention: 365d")


def setup_continuous_query():
    """
    创建降采样任务（InfluxDB v2叫Task）
    每5分钟执行一次：将1分钟粒度聚合为1小时粒度
    """
    task_flux = '''
    option task = {{
      name: "downsample-1h",
      every: 5m
    }}

    from(bucket: "sensor_data")
      |> range(start: -task.every)
      |> filter(fn: (r) => r._measurement == "sensor_data")
      |> aggregateWindow(every: 1h, fn: mean, createEmpty: false)
      |> to(bucket: "sensor_downsampled", org: "digital_twin")
    '''.format()
    
    # 在InfluxDB Web界面: Data > Tasks > Create > 粘贴上述Flux
    print("[配置] 请在InfluxDB Web界面创建Task:")
    print("  Data > Tasks > Create Task")
    print("  粘贴降采样Flux脚本")
    print("  1分钟粒度 → 1小时粒度（mean聚合）")


# ====== 集成到边缘节点 ======
def integrated_edge_node():
    """
    将InfluxDB写入集成到上周的edge_node.py
    在MQTT回调中添加InfluxDB写入
    """
    integration_code = '''
# 在 edge_node.py 的 on_sensor_data 函数中添加：

def on_sensor_data(client, userdata, msg):
    """收到传感器数据 → 卡尔曼滤波 → MQTT发布 → InfluxDB写入"""
    try:
        raw = json.loads(msg.payload.decode())
        # 卡尔曼滤波（已有代码）
        filtered = {k: filters[k].update(v) for k, v in raw.items() if k in filters}
        
        # MQTT发布滤波数据（已有代码）
        client.publish(TOPIC_SENSOR, json.dumps(filtered))
        
        # ★ 新增：写入InfluxDB ★
        write_sensor_data(filtered)
        
    except Exception as e:
        print(f"处理失败: {e}")
'''
    print(integration_code)


if __name__ == "__main__":
    # 测试写入
    test_data = {"t": 25.3, "h": 60.2, "p": 1013.2, "d": 15.6}
    write_sensor_data(test_data)
    
    # 查询测试
    time.sleep(1)
    history = query_history(field="temperature", window="1h", duration="-1h")
    print(f"历史数据: {history}")
    
    print("\n===== 部署指南 =====")
    setup_retention_policy()
    setup_continuous_query()
    integrated_edge_node()
```

### 2. 算法题（20分钟）

**题目：时序数据异常检测 — Z-Score滑动窗口**

```
给定一个温度时序数据流，用滑动窗口计算Z-Score，
当 |Z-Score| > 3 时标记为异常点。

输入：[25.1, 25.3, 25.2, 25.4, 25.3, 35.0, 25.2, 25.1]
窗口大小：5
输出：异常点的索引列表

要求：
1. 实现滑动窗口Z-Score计算
2. 时间复杂度O(n)，空间复杂度O(window_size)
3. 用增量计算（不重复计算整个窗口的均值和方差）

提示：维护窗口内的sum和sum_of_squares，用Welford算法增量更新
```

```python
def sliding_zscore(data, window=5, threshold=3.0):
    """
    滑动窗口Z-Score异常检测
    
    Args:
        data: 时序数据列表
        window: 滑动窗口大小
        threshold: Z-Score阈值
    
    Returns:
        anomalies: [(index, value, zscore), ...]
    """
    n = len(data)
    anomalies = []
    
    # 增量统计量
    w_sum = 0.0
    w_sq_sum = 0.0
    
    for i in range(n):
        x = data[i]
        w_sum += x
        w_sq_sum += x * x
        
        # 窗口未满时移除旧数据
        if i >= window:
            old = data[i - window]
            w_sum -= old
            w_sq_sum -= old * old
        
        # 计算当前窗口统计量
        count = min(i + 1, window)
        if count < 2:
            continue
        
        mean = w_sum / count
        variance = (w_sq_sum / count) - (mean * mean)
        std = variance ** 0.5 if variance > 0 else 0
        
        # Z-Score
        if std > 0:
            zscore = abs((x - mean) / std)
            if zscore > threshold:
                anomalies.append((i, x, zscore))
    
    return anomalies


# 测试
data = [25.1, 25.3, 25.2, 25.4, 25.3, 35.0, 25.2, 25.1]
result = sliding_zscore(data, window=5, threshold=3.0)
for idx, val, z in result:
    print(f"异常: 索引{idx}, 值={val}, Z-Score={z:.2f}")
# 预期输出: 异常: 索引5, 值=35.0, Z-Score≈3.27
```
