### 卡尔曼滤波核心原理

卡尔曼滤波是一种**递归最优估计**算法，在数字孪生中用于：
- 消除传感器噪声
- 多源数据融合
- 实时状态预测

**核心公式**（只需理解思想，不需背公式）：

```
预测阶段：基于物理模型预测下一状态
    ↓
更新阶段：用实际测量值校正预测值
```

**参数调优直觉**：

| 参数 | 增大 | 减小 |
|------|------|------|
| **Q (过程噪声)** | 更信任观测 | 更信任预测 |
| **R (观测噪声)** | 更信任预测 | 更信任观测 |

### 1. 基础卡尔曼滤波器（Python实现）

```python
import numpy as np
import matplotlib.pyplot as plt

class KalmanFilter:
    """
    一维卡尔曼滤波器 - 适用于传感器去噪
    状态向量：[位置, 速度]
    """
    def __init__(self, dt=0.1, process_noise=0.01, measurement_noise=0.04):
        self.dt = dt  # 采样间隔
        # 状态转移矩阵 (匀速模型)
        self.F = np.array([[1, dt], 
                           [0, 1]])
        # 观测矩阵 (只观测位置)
        self.H = np.array([[1, 0]])
        # 过程噪声协方差
        self.Q = np.eye(2) * process_noise
        # 观测噪声协方差
        self.R = np.array([[measurement_noise]])
        # 初始状态 [位置, 速度]
        self.x = np.array([[0], [0]])
        # 初始协方差
        self.P = np.eye(2)
    
    def predict(self):
        """预测阶段：基于模型预测"""
        self.x = self.F @ self.x
        self.P = self.F @ self.P @ self.F.T + self.Q
        return self.x
    
    def update(self, z):
        """更新阶段：用测量值校正"""
        # 计算卡尔曼增益
        S = self.H @ self.P @ self.H.T + self.R
        K = self.P @ self.H.T @ np.linalg.inv(S)
        
        # 状态更新
        y = z - self.H @ self.x  # 残差
        self.x = self.x + K @ y
        
        # 协方差更新
        I = np.eye(2)
        self.P = (I - K @ self.H) @ self.P
        
        return self.x[0, 0]  # 返回滤波后的位置

def test_kalman_filter():
    """测试卡尔曼滤波效果"""
    # 生成测试数据：带噪声的正弦信号
    np.random.seed(42)
    t = np.linspace(0, 10, 200)
    true_signal = np.sin(t)  # 真实信号
    noise = np.random.normal(0, 0.3, len(t))  # 高斯噪声
    measurements = true_signal + noise
    
    # 创建滤波器
    kf = KalmanFilter(dt=0.05, process_noise=0.001, measurement_noise=0.09)
    
    # 滤波处理
    filtered = []
    for z in measurements:
        kf.predict()
        filtered.append(kf.update(np.array([[z]])))
    
    # 可视化对比
    plt.figure(figsize=(12, 4))
    plt.plot(t, measurements, 'r.', alpha=0.3, label='Raw Measurements')
    plt.plot(t, true_signal, 'g-', linewidth=2, label='True Signal')
    plt.plot(t, filtered, 'b-', linewidth=2, label='Kalman Filtered')
    plt.legend()
    plt.title('Kalman Filter: Noise Reduction Effect')
    plt.savefig('kalman_demo.png', dpi=150)
    plt.show()

if __name__ == '__main__':
    test_kalman_filter()
```

### 2. 传感器数据预处理管道

```python
import serial
import json
from kalman_filter import KalmanFilter

class SensorDataPipeline:
    """传感器数据预处理管道 - 适用于你的DHT11+HC-SR04组合"""
    
    def __init__(self):
        # 为温度和距离各创建滤波器
        self.temp_filter = KalmanFilter(dt=1.0, process_noise=0.01, measurement_noise=0.25)
        self.dist_filter = KalmanFilter(dt=0.1, process_noise=0.1, measurement_noise=0.04)
        
        self.buffer = {
            'temperature': [],
            'humidity': [],
            'distance': [],
            'timestamp': []
        }
    
    def process_serial_data(self, raw_data):
        """
        处理串口数据 (Arduino发送的JSON格式)
        raw_data: {"temp": 25.3, "hum": 60.2, "dist": 45.7}
        """
        # 解析数据
        data = json.loads(raw_data)
        
        # 卡尔曼滤波
        temp_filtered = self.temp_filter.update(np.array([[data['temp']]]))
        dist_filtered = self.dist_filter.update(np.array([[data['dist']]]))
        
        # 构建处理后的数据
        processed = {
            'temperature': round(temp_filtered, 2),
            'humidity': data['hum'],  # 湿度波动小，可不滤波
            'distance': round(dist_filtered, 2),
            'timestamp': data.get('ts', None)
        }
        
        # 更新缓冲区
        for key in ['temperature', 'humidity', 'distance']:
            self.buffer[key].append(processed[key])
            # 保留最近100个数据点
            if len(self.buffer[key]) > 100:
                self.buffer[key] = self.buffer[key][-100:]
        
        return processed
    
    def get_statistics(self):
        """计算滑动窗口统计量"""
        return {
            'temp_mean': np.mean(self.buffer['temperature'][-10:]),
            'temp_std': np.std(self.buffer['temperature'][-10:]),
            'dist_mean': np.mean(self.buffer['distance'][-10:]),
            'dist_std': np.std(self.buffer['distance'][-10:])
        }
```

### 4. MQTT-WebSocket桥接服务（Node.js）

```javascript
// mqtt_websocket_bridge.js
const mqtt = require('mqtt');
const WebSocket = require('ws');

class MQTTSocketBridge {
    constructor(mqttBroker, wsPort = 8080) {
        this.mqttClient = mqtt.connect(mqttBroker);
        this.wsServer = new WebSocket.Server({ port: wsPort });
        
        this.setupMQTT();
        this.setupWebSocket();
    }
    
    setupMQTT() {
        this.mqttClient.on('connect', () => {
            console.log('MQTT Broker已连接');
            // 订阅传感器数据主题
            this.mqttClient.subscribe('sensors/#', (err) => {
                if (!err) console.log('已订阅 sensors/#');
            });
        });
        
        this.mqttClient.on('message', (topic, message) => {
            // 广播给所有WebSocket客户端
            const data = JSON.parse(message.toString());
            this.broadcast({ topic, ...data });
        });
    }
    
    setupWebSocket() {
        this.wsServer.on('connection', (ws) => {
            console.log('WebSocket客户端已连接');
            
            ws.on('message', (message) => {
                // 处理客户端发送的控制指令
                const cmd = JSON.parse(message.toString());
                if (cmd.type === 'control') {
                    this.mqttClient.publish('actuator/' + cmd.device, 
                        JSON.stringify(cmd.data));
                }
            });
            
            ws.on('close', () => {
                console.log('WebSocket客户端断开');
            });
        });
    }
    
    broadcast(data) {
        this.wsServer.clients.forEach((client) => {
            if (client.readyState === WebSocket.OPEN) {
                client.send(JSON.stringify(data));
            }
        });
    }
}

// 启动桥接服务
```

```cpp
#include <DHT.h>
#include <NewPing.h>

#define DHT_PIN 2
#define TRIG_PIN 9
#define ECHO_PIN 10
#define DHTTYPE DHT11

DHT dht(DHT_PIN, DHTTYPE);
NewPing sonar(TRIG_PIN, ECHO_PIN, 200);

unsigned long lastSend = 0;
const int INTERVAL = 500; // 500ms

void setup() {
    Serial.begin(115200);
    dht.begin();
}

void loop() {
    if (millis() - lastSend >= INTERVAL) {
        float h = dht.readHumidity();
        float t = dht.readTemperature();
        int d = sonar.ping_cm();
        
        // JSON格式输出
        Serial.print("{\"temp\":");
        Serial.print(t);
        Serial.print(",\"hum\":");
        Serial.print(h);
        Serial.print(",\"dist\":");
        Serial.print(d);
        Serial.println("}");
        
        lastSend = millis();
    }
}
```

```python
# raspberry_sensor_processor.py
import serial
import paho.mqtt.client as mqtt
import json
from kalman_filter import KalmanFilter

class SensorNode:
    def __init__(self):
        # 串口连接 (根据实际端口调整)
        self.ser = serial.Serial('/dev/ttyACM0', 115200, timeout=1)
        
        # MQTT客户端
        self.mqtt = mqtt.Client()
        self.mqtt.connect('localhost', 1883)
        self.mqtt.loop_start()
        
        # 卡尔曼滤波器
        self.temp_filter = KalmanFilter(dt=0.5, process_noise=0.1, measurement_noise=0.25)
        self.dist_filter = KalmanFilter(dt=0.5, process_noise=0.1, measurement_noise=0.04)
    
    def run(self):
        print("开始读取传感器数据...")
        try:
            while True:
                if self.ser.in_waiting:
                    line = self.ser.readline().decode('utf-8').strip()
                    try:
                        data = json.loads(line)
                        
                        # 卡尔曼滤波
                        temp_filtered = self.temp_filter.update(
                            float(data.get('temp', 0)))
                        dist_filtered = self.dist_filter.update(
                            float(data.get('dist', 0)))
                        
                        # 构建滤波后的数据
                        output = {
                            'temperature': round(temp_filtered, 2),
                            'humidity': data.get('hum', 0),
                            'distance': round(dist_filtered, 2),
                            'raw_temp': data.get('temp', 0),
                            'raw_dist': data.get('dist', 0)
                        }
                        
                        # 发布到MQTT
                        self.mqtt.publish('sensors/dht11_hc-sr04', 
                                         json.dumps(output))
                        
                        print(f"发送: {output}")
                        
                    except json.JSONDecodeError:
                        pass  # 忽略无效行
                        
        except KeyboardInterrupt:
            print("\n停止采集")
            self.ser.close()
            self.mqtt.loop_stop()

if __name__ == '__main__':
    node = SensorNode()
    node.run()
```
