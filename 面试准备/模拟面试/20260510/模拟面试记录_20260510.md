## 一、算法环节（30分钟）

### 题目1：LeetCode 70 - 爬楼梯（简单）

**问题**：假设你正在爬楼梯。需要 n 阶你才能到达楼顶。每次你可以爬 1 或 2 个台阶。有多少种不同的方法可以爬到楼顶？

**面试思考过程**：

```
思路分析：
- 这是一个经典的动态规划问题
- 到达第n阶的方法 = 到达第n-1阶 + 到达第n-2阶
- 因为最后一步要么从n-1爬1步，要么从n-2爬2步

状态定义：dp[i] = 爬到第i阶的方法数
状态转移：dp[i] = dp[i-1] + dp[i-2]
初始化：dp[0] = 1, dp[1] = 1
```

**代码实现**：

```cpp
class Solution {
public:
    int climbStairs(int n) {
        if (n <= 2) return n;
        
        int prev1 = 1;  // dp[i-1]
        int prev2 = 1;  // dp[i-2]
        int current = 0;
        
        for (int i = 2; i <= n; ++i) {
            current = prev1 + prev2;
            prev2 = prev1;
            prev1 = current;
        }
        
        return prev1;
    }
};

// 空间优化版本：只需O(1)空间
```

**复杂度分析**：
- 时间复杂度：O(n)
- 空间复杂度：O(1)

**追问**：能否用矩阵快速幂优化到O(log n)？
```cpp
// 快速幂解法
class Solution {
public:
    int climbStairs(int n) {
        if (n <= 2) return n;
        
        // 矩阵乘法
        vector<vector<long long>> base = {{1,1},{1,0}};
        vector<vector<long long>> res = matrixPower(base, n-1);
        return res[0][0] + res[1][0];
    }
};
```

### 题目2：LeetCode 200 - 岛屿数量（中等）

**问题**：给你一个由 '1'（陆地）和 '0'（水）组成的二维网格，请你计算网格中岛屿的数量。岛屿总是被水包围，并且每座岛屿只能由水平或垂直方向上相邻的陆地连接形成。

**解题思路**：

```cpp
class Solution {
public:
    int numIslands(vector<vector<char>>& grid) {
        if (grid.empty()) return 0;
        
        int m = grid.size();
        int n = grid[0].size();
        int islands = 0;
        
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (grid[i][j] == '1') {
                    ++islands;
                    bfs(grid, i, j);
                }
            }
        }
        return islands;
    }
    
private:
    void bfs(vector<vector<char>>& grid, int i, int j) {
        queue<pair<int,int>> q;
        q.push({i,j});
        grid[i][j] = '0';  // 标记已访问
        
        vector<pair<int,int>> dirs = {{0,1},{0,-1},{1,0},{-1,0}};
        
        while (!q.empty()) {
            auto [x,y] = q.front();
            q.pop();
            
            for (auto [dx,dy] : dirs) {
                int nx = x + dx;
                int ny = y + dy;
                
                if (nx >= 0 && nx < grid.size() && 
                    ny >= 0 && ny < grid[0].size() && 
                    grid[nx][ny] == '1') {
                    grid[nx][ny] = '0';
                    q.push({nx,ny});
                }
            }
        }
    }
};
```

**复杂度**：
- 时间：O(m×n)
- 空间：O(min(m×n, BFS队列深度))

```cpp
UCLASS()
class MUSEUM_API UTeleportManager : public UObject
{
    GENERATED_BODY()
    
public:
    // 触发传送
    UFUNCTION(BlueprintCallable)
    void InitiateTeleport(APlayerController* Player, 
                          FVector TargetLocation, 
                          FRotator TargetRotation);
    
protected:
    // 查找有效传送点
    FVector FindValidSpawnPoint(UWorld* World, 
                                 const FVector& DesiredLocation);
    
    // 播放传送特效
    void PlayTransitionEffect(APawn* Pawn, bool bIsExit);
    
    // 处理网络同步
    UFUNCTION(Server, Reliable)
    void Server_TeleportPlayer(APlayerController* Player,
                               FVector NewLocation,
                               FRotator NewRotation);
};

void UTeleportManager::InitiateTeleport(APlayerController* Player,
                                        FVector TargetLocation,
                                        FRotator TargetRotation)
{
    if (!Player) return;
    
    APawn* Pawn = Player->GetPawn();
    if (!Pawn) return;
    
    // 1. 禁用输入
    Player->DisableInput(nullptr);
    
    // 2. 播放传送特效（进入传送门）
    PlayTransitionEffect(Pawn, false);
    
    // 3. 延迟执行传送
    FTimerHandle TimerHandle;
    GetWorld()->GetTimerManager().SetTimer(TimerHandle, [this, Player, TargetLocation, TargetRotation]()
    {
        // 4. 计算有效传送点
        FVector SpawnPoint = FindValidSpawnPoint(GetWorld(), TargetLocation);
        
        // 5. 移动玩家
        FRotator SafeRotation = TargetRotation;
        
        // 6. 网络同步
        if (HasAuthority())
        {
            Server_TeleportPlayer(Player, SpawnPoint, SafeRotation);
        }
        
        // 7. 播放传送特效（离开传送门）
        PlayTransitionEffect(Player->GetPawn(), true);
        
        // 8. 恢复输入
        Player->EnableInput(nullptr);
    }, 1.5f, false);
}
```

```cpp
FVector UTeleportManager::FindValidSpawnPoint(UWorld* World, 
                                              const FVector& DesiredLocation)
{
    FVector SpawnPoint = DesiredLocation;
    
    // 尝试多次检测避免传送到墙里
    for (int32 i = 0; i < 10; ++i)
    {
        // 使用胶囊体检测
        FCollisionQueryParams QueryParams;
        QueryParams.AddIgnoredActor(GetOwner());
        
        FHitResult HitResult;
        bool bHit = World->LineTraceSingleByChannel(
            HitResult,
            SpawnPoint + FVector(0, 0, 100),  // 从上方检测
            SpawnPoint - FVector(0, 0, 100),
            ECC_Visibility,
            QueryParams
        );
        
        if (bHit)
        {
            // 如果碰撞到地面，上移50单位
            SpawnPoint = HitResult.ImpactPoint + FVector(0, 0, 50.f);
        }
        else
        {
            // 安全位置
            break;
        }
    }
    
    return SpawnPoint;
}
```

#### 3. 空间划分策略

```cpp
// 使用Grid-based空间划分
void UWorldStreamingManager::InitializeGrid(FIntVector WorldSize, int32 ChunkSize)
{
    GridSize = ChunkSize;
    
    // 创建Chunk索引
    int32 ChunkCountX = WorldSize.X / ChunkSize;
    int32 ChunkCountY = WorldSize.Y / ChunkSize;
    int32 ChunkCountZ = WorldSize.Z / ChunkSize;
    
    ChunkGrid.Empty();
    ChunkGrid.Reserve(ChunkCountX * ChunkCountY * ChunkCountZ);
    
    for (int32 z = 0; z < ChunkCountZ; ++z)
    {
        for (int32 y = 0; y < ChunkCountY; ++y)
        {
            for (int32 x = 0; x < ChunkCountX; ++x)
            {
                FIntVector ChunkCoord(x, y, z);
                FVector ChunkCenter = FVector(
                    x * ChunkSize + ChunkSize/2,
                    y * ChunkSize + ChunkSize/2,
                    z * ChunkSize + ChunkSize/2
                );
                
                FStreamingChunk* NewChunk = new FStreamingChunk();
                NewChunk->Coord = ChunkCoord;
                NewChunk->Center = ChunkCenter;
                NewChunk->bIsLoaded = false;
                NewChunk->AssetPath = GenerateChunkAssetPath(ChunkCoord);
                
                ChunkGrid.Add(NewChunk);
            }
        }
    }
}
```

#### 4. 加载优先级计算

```cpp
struct FChunkPriority
{
    FIntVector Coord;
    float Priority;
    float Distance;
    EChunkLOD TargetLOD;
};

TArray<FChunkPriority> UWorldStreamingManager::CalculatePriorities(
    const FVector& PlayerLocation,
    const FVector& PlayerVelocity,
    float DeltaTime)
{
    TArray<FChunkPriority> Priorities;
    
    // 预测位置
    FVector PredictedLocation = PlayerLocation + PlayerVelocity * PredictionTime;
    
    for (FStreamingChunk* Chunk : ChunkGrid)
    {
        // 计算到当前玩家位置的距离
        float DistanceCurrent = FVector::Dist(PlayerLocation, Chunk->Center);
        
        // 计算到预测位置的距离
        float DistancePredict = FVector::Dist(PredictedLocation, Chunk->Center);
        
        // 取较小值（考虑预加载）
        float Distance = FMath::Min(DistanceCurrent, DistancePredict);
        
        // 计算LOD等级
        EChunkLOD TargetLOD = CalculateLOD(Distance);
        
        // 优先级 = 10000（已加载需保持） + (1/距离) * 权重
        float Priority = 0.f;
        
        if (!Chunk->bIsLoaded)
        {
            // 未加载Chunk：距离越近优先级越高
            Priority = 10000.f / FMath::Max(Distance, 1.f);
        }
        else if (Chunk->CurrentLOD != TargetLOD)
        {
            // 需要LOD切换
            Priority = 5000.f / FMath::Max(Distance, 1.f);
        }
        
        // 方向预测加权
        FVector ToChunk = (Chunk->Center - PlayerLocation).GetSafeNormal();
        FVector PlayerDir = PlayerVelocity.GetSafeNormal();
        float DirectionBonus = FVector::DotProduct(ToChunk, PlayerDir);
        
        if (DirectionBonus > 0)
        {
            Priority *= (1.f + DirectionBonus);
        }
        
        Priorities.Add({Chunk->Coord, Priority, Distance, TargetLOD});
    }
    
    // 按优先级排序
    Priorities.Sort([](const FChunkPriority& A, const FChunkPriority& B)
    {
        return A.Priority > B.Priority;
    });
    
    return Priorities;
}
```

#### 5. 异步加载策略

```cpp
void UWorldStreamingManagement::ProcessStreamingQueue()
{
    // 获取优先级列表
    TArray<FChunkPriority> Priorities = CalculatePriorities(
        GetPlayerLocation(),
        GetPlayerVelocity(),
        DeltaTime
    );
    
    int32 LoadCounter = 0;
    int32 MaxConcurrentLoads = 4;  // 最多同时加载4个
    
    for (const FChunkPriority& P : Priorities)
    {
        if (LoadCounter >= MaxConcurrentLoads) break;
        
        FStreamingChunk* Chunk = GetChunk(P.Coord);
        if (!Chunk) continue;
        
        // 卸载过远的Chunk
        if (Chunk->bIsLoaded && P.Distance > UnloadRadius)
        {
            StreamOutChunk(Chunk);
            continue;
        }
        
        // 加载需要的Chunk
        if (!Chunk->bIsLoaded && P.Distance < LoadRadius)
        {
            StreamInChunk(Chunk);
            LoadCounter++;
        }
        
        // LOD切换
        if (Chunk->bIsLoaded && Chunk->CurrentLOD != P.TargetLOD)
        {
            SwitchChunkLOD(Chunk, P.TargetLOD);
        }
    }
}
```
